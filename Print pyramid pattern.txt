#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {

    int n,i,j,k;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        for(j=n;j>i;j--)
        {
           printf(" ");
        }
        for(k=1;k<=2*i-1;k++)
        {
            printf("*");
           
        }
        
        printf("\n");
    }
}
